void desplazamiento(int tecla, int &x, int &y)
{
	if (_kbhit)
	{
		tecla = _getch();
		if (tecla = 224)
		{
			Console::SetCursorPosition(x + 2, y);
			cout << " ";
			Console::SetCursorPosition(x + 1, y + 1);
			cout << " "<<" " << " ";
			tecla = getch();
			switch (tecla)
			{
				
			case DERECHA:
					x++;
				break;
			case IZQUIERDA:
					x--;
				break;
			case ARRIBA:
					y--;
				break;
			case ABAJO:
					y++;
				break;
			}
			rana();
		}
	}
}
