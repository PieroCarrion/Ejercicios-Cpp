bool Colision(int x, int y, int ex, int ey, int ex2, int ey2, int ex3, int ey3)
{
	if (x >= ex&& x <= ex + 2)
	{
		if (y >= ey && y <= ey + 2)return true;
	}
	if (x >= ex2&& x <= ex2 + 2)
	{
		if (y >= ey && y <= ey + 2)return true;
	}
	if (x >= ex3&& x <= ex3 + 2)
	{
		if (y >= ey && y <= ey + 2)return true;
	}
	return false;
}
void colorFondo(int x, int y)  // para no dejar rastro
{
	switch (mapa[y][x])
	{
	case 0:
		Console::BackgroundColor = ConsoleColor::Cyan; break;
	case 1:
		Console::BackgroundColor = ConsoleColor::Red; break;
	case 2:
		Console::BackgroundColor = ConsoleColor::Black; break;
	case 3:
		Console::BackgroundColor = ConsoleColor::Green; break;
	case 4:
		Console::BackgroundColor = ConsoleColor::Gray; break;

	}

}
void gotoxy(int x, int y) {
	Console::SetCursorPosition(x, y);
	colorFondo(x, y);
}
void dibujarMapa() {

	//LLENA EL MAPA PRINCIPAL
	for (int i = 0; i < D; i++)
	{
		for (int j = 0; j < D; j++)
		{
			switch (numMapa)
			{
			case 1: mapa[i][j] = mapa1[i][j]; break;
			case 2: mapa[i][j] = mapa2[i][j]; break;
			case 3: mapa[i][j] = mapa3[i][j]; break;
			case 4: mapa[i][j] = mapa4[i][j]; break;
			case 5: mapa[i][j] = mapa5[i][j]; break;
			}
		}
	}

	//DIBUJAR CADA COORDENADAD
	for (int i = 0; i < D; i++)
	{
		for (int j = 0; j < D; j++)
		{
			gotoxy(j, i);
			cout << " ";
		}
	}
}

void cambiardeMapa(int &x, int &y) {
	if (y == 2) {
		if (numMapa == 1)numMapa = 2;
		else if (numMapa == 2)numMapa = 3;
		else if (numMapa == 3)numMapa = 4;

		y = Y0;
	}
}
void juego(int &x, int &y, int &ex, int &ey, int &dx, int &ex2, int &ey2, int &dx2, int &ex3, int &ey3, int &dx3)
{

	dibujarPersona(x, y);
	char tecla = -1;
	dibujae(ex, ey);
	dibujae2(ex2, ey2);
	dibujae3(ex3, ey3);
	while (true)
	{
		_sleep(100);
		eliminare(ex, ey);
		eliminare2(ex2, ey2);
		eliminare3(ex3, ey3);
		if (kbhit())
		{
			tecla = getch();
			borrarPersona(x, y);
			switch (tecla)
			{

			case DERECHA: case DERECHA1: if ((mapa[y][x] != 3 && x<D - 1) && (mapa[y + 1][x] != 3 && x<D - 1))
				x++; break;
			case IZQUIERDA: case IZQUIERDA1:if ((mapa[y][x - 1] != 3 && x>0) && (mapa[y + 1][x - 1] != 3 && x>0))
				x--; break;
			case ARRIBA: case ARRIBA1:if ((mapa[y - 1][x] != 3) && (mapa[y - 1][x + 1] != 3) && (mapa[y - 1][x + 2] != 3) && (mapa[y - 1][x + 3] != 3))
				y--; break;
			case ABAJO: case ABAJO1:if ((mapa[y + 2][x] != 3) && (mapa[y + 2][x + 1] != 3) && (mapa[y + 2][x + 2] != 3) && (mapa[y + 2][x + 3] != 3))
				y++; break;
			}
			cambiardeMapa(x, y);
			if (Colision(x, y, ex, ey, ex2, ey2, ex3, ey3)) {
				x = X0; y = Y0;
			}
			dibujarPersona(x, y);

		}


		moverenemigo2(ex2, dx2);
		moverenemigo3(ex3, dx3);
		moverenemigo(ex, dx);
		dibujae2(ex2, ey2);
		dibujae3(ex3, ey3);
		dibujae(ex, ey);


	}

}

	int ex3 = 18, ey3 = 18, dx3 = 1;
	dibujarMapa();
