#include "stdio.h"
#include "conio.h"
#include "iostream"

using namespace std;

int main()
{
	int n;
	int espacios;
	cout<<"Ingrese n: ";
	cin>>n;
	espacios=n-1;

	for (int i = 0; i < n; i++,cout<<"\n")
	{
		for (int j = 0; j < n; j++)
		{
			if(j<espacios)
			{
				cout<<" ";
			}
			else
				cout<<"*";
		}
		espacios--;
	}

	_getch();

}
